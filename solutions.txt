select d.dept_id, d.dept_name, coalesce(sum(e.salary),0) as total from department d left join employee e 
on  d.dept_id = e.dept_id 
group by d.dept_id,d.dept_name;



select e.dept_id, coalesce(d.dept_name,'others') as dept_name, sum(e.salary) as total from department d right join employee e 
on  d.dept_id = e.dept_id 
group by d.dept_id,d.dept_name;


create table emp(emp_id integer(4) primary key,emp_name varchar(20),manager_id integer(4),salary integer(4));
desc emp;

select e.emp_id as manager_id,e.emp_name as manager,sum(m.salary) 
from emp e  join emp m on e.emp_id = m.manager_id
 group by manager_id,manager;


with result as (
select emp_id,emp_name,salary,dense_rank() over(order by salary desc) ra from emp
